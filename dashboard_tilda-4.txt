<script>
(function(){
  const API_BASE = "https://Barluch.pythonanywhere.com";
  const DASH_KEY = "ass112345jdbfsjdhf873hsdjfh";

  const $ = (id) => document.getElementById(id);

  const elStatus = $("ymStatus");
  const elPeriod = $("ymPeriod");
  const elReload = $("ymReload");

  const elKpiCreated = $("kpiCreated");
  const elKpiReserved = $("kpiReserved");
  const elKpiRatio = $("kpiRatio");
  const elKpiPeak = $("kpiPeak");

  const elKpiCreatedHint = $("kpiCreatedHint");
  const elKpiReservedHint = $("kpiReservedHint");
  const elKpiRatioHint = $("kpiRatioHint");
  const elKpiPeakHint = $("kpiPeakHint");

  const elMetaRange = $("metaRange");
  const elMetaSegments = $("metaSegments");
  const elMetaHourly = $("metaHourly");
  const elSummaryList = $("summaryList");
  const elSegmentTableBody = $("segmentTableBody");

  const ctxTimeline = $("chartTimeline");
  const ctxSegments = $("chartSegments");
  const ctxHourly = $("chartHourly");

  let chartTimeline = null;
  let chartSegments = null;
  let chartHourly = null;

  function setStatus(text){
    elStatus.textContent = text;
  }

  function qs(params){
    return new URLSearchParams(params).toString();
  }

  async function apiGet(path, params){
    const url = API_BASE + path + "?" + qs({ ...params, key: DASH_KEY });
    const res = await fetch(url, { method: "GET" });
    if (!res.ok) throw new Error("HTTP " + res.status);
    return await res.json();
  }

  function destroyChart(ch){
    if (ch && typeof ch.destroy === "function") ch.destroy();
  }

  function safeNum(v){
    const n = Number(v);
    return Number.isFinite(n) ? n : 0;
  }

  function formatInt(v){
    return new Intl.NumberFormat("ru-RU").format(safeNum(v));
  }

  function formatPct(v){
    const n = Math.round(safeNum(v) * 10) / 10;
    return (Number.isInteger(n) ? String(n) : String(n).replace(".", ",")) + "%";
  }

  function sum(arr){
    return (arr || []).reduce((acc, x) => acc + safeNum(x), 0);
  }

  function maxIndex(arr){
    if (!arr || !arr.length) return -1;
    let idx = 0;
    let max = safeNum(arr[0]);
    for (let i = 1; i < arr.length; i++){
      const v = safeNum(arr[i]);
      if (v > max){
        max = v;
        idx = i;
      }
    }
    return idx;
  }

  function humanPeriodLabel(key){
    if (key === "today") return "Сегодня";
    if (key === "yesterday") return "Вчера";
    if (key === "week") return "Последние 7 дней";
    if (key === "30d") return "Последние 30 дней";
    return key || "—";
  }

  function toRuDateTime(str){
    if (!str) return "—";
    const normalized = String(str).replace(" ", "T");
    const d = new Date(normalized);
    if (isNaN(d.getTime())) return str;
    return d.toLocaleString("ru-RU");
  }

  function shortLabel(label){
    if (!label) return "";
    if (label.length > 16 && label.indexOf(" ") > -1){
      return label.slice(5);
    }
    return label;
  }

  function renderSummary(items){
    elSummaryList.innerHTML = items.map(item => (
      '<div class="ymdash-summary__item">' +
        '<div class="ymdash-summary__title">' + item.title + '</div>' +
        '<div class="ymdash-summary__text">' + item.text + '</div>' +
      '</div>'
    )).join("");
  }

  function renderSegmentTable(buckets){
    if (!buckets || !buckets.length){
      elSegmentTableBody.innerHTML = '<tr><td colspan="3">Нет данных</td></tr>';
      return;
    }

    elSegmentTableBody.innerHTML = buckets.map(b => (
      "<tr>" +
        "<td><span class='ymdash-pill'>" + escapeHtml(b.label || "—") + "</span></td>" +
        "<td>" + formatInt(b.count) + "</td>" +
        "<td>" + formatPct(b.pct) + "</td>" +
      "</tr>"
    )).join("");
  }

  function escapeHtml(str){
    return String(str)
      .replaceAll("&", "&amp;")
      .replaceAll("<", "&lt;")
      .replaceAll(">", "&gt;")
      .replaceAll('"', "&quot;")
      .replaceAll("'", "&#039;");
  }

  function chartOptionsBase(){
    return {
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: {
          position: "bottom",
          labels: {
            color: "#475467",
            boxWidth: 14,
            usePointStyle: true,
            pointStyle: "circle",
            padding: 18,
            font: { size: 12, weight: "600" }
          }
        },
        tooltip: {
          backgroundColor: "#101828",
          titleColor: "#ffffff",
          bodyColor: "#ffffff",
          padding: 10,
          displayColors: true
        }
      },
      scales: {
        x: {
          ticks: { color: "#667085", maxRotation: 0, autoSkip: true },
          grid: { color: "rgba(16,24,40,0.05)" }
        },
        y: {
          beginAtZero: true,
          ticks: { color: "#667085" },
          grid: { color: "rgba(16,24,40,0.07)" }
        }
      }
    };
  }

  async function loadData(){
    const period = (elPeriod.value || "today").trim();

    setStatus("Загружаю данные…");
    elReload.disabled = true;

    try{
      const [segments, load] = await Promise.all([
        apiGet("/admin/api/segments", {}),
        apiGet("/admin/api/load", { period })
      ]);

      if (!segments?.ok) throw new Error("Bad payload: segments");
      if (!load?.ok) throw new Error("Bad payload: load");

      const totals = load.totals || {};
      const timeseries = load.timeseries || {};
      const hourly = load.hourly || {};
      const p = load.period || {};

      const createdTotal = safeNum(totals.created_total);
      const reservedTotal = safeNum(totals.reserved_total);
      const ratio = createdTotal > 0 ? (reservedTotal / createdTotal * 100) : 0;

      const peakIdx = maxIndex(hourly.reserved || []);
      const peakHourLabel = peakIdx >= 0 ? (hourly.labels?.[peakIdx] || "—") : "—";
      const peakHourValue = peakIdx >= 0 ? safeNum(hourly.reserved?.[peakIdx]) : 0;

      elKpiCreated.textContent = formatInt(createdTotal);
      elKpiReserved.textContent = formatInt(reservedTotal);
      elKpiRatio.textContent = formatPct(ratio);
      elKpiPeak.textContent = peakHourLabel === "—" ? "—" : (peakHourLabel + ":00");

      elKpiCreatedHint.textContent = "События CREATED за период: " + humanPeriodLabel(period);
      elKpiReservedHint.textContent = "Брони с reservation_dt за период: " + humanPeriodLabel(period);
      elKpiRatioHint.textContent = "Фактические / созданные: " + formatInt(reservedTotal) + " из " + formatInt(createdTotal);
      elKpiPeakHint.textContent = peakHourValue > 0
        ? ("Максимум фактических броней: " + formatInt(peakHourValue))
        : "Нет выраженного пика";

      elMetaRange.textContent =
        humanPeriodLabel(period) +
        " • " +
        toRuDateTime(p.from) +
        " — " +
        toRuDateTime(p.to_exclusive);

      elMetaSegments.textContent =
        "Активные гости за 2 года • всего: " + formatInt(segments.total_guests || 0);

      const reservedHourSeries = hourly.reserved || [];
      const createdHourSeries = hourly.created || [];
      const labelsHour = hourly.labels || [];

      const createdTimeline = timeseries.created || [];
      const reservedTimeline = timeseries.reserved || [];
      const labelsTimeline = (timeseries.labels || []).map(shortLabel);

      const timelinePeakIdx = maxIndex(reservedTimeline);
      const timelinePeakLabel = timelinePeakIdx >= 0 ? (timeseries.labels?.[timelinePeakIdx] || "—") : "—";
      const timelinePeakValue = timelinePeakIdx >= 0 ? safeNum(reservedTimeline?.[timelinePeakIdx]) : 0;

      const createPeakIdx = maxIndex(createdTimeline);
      const createPeakLabel = createPeakIdx >= 0 ? (timeseries.labels?.[createPeakIdx] || "—") : "—";
      const createPeakValue = createPeakIdx >= 0 ? safeNum(createdTimeline?.[createPeakIdx]) : 0;

      elMetaHourly.textContent =
        peakHourValue > 0
          ? ("Пиковый час фактических броней: " + peakHourLabel + ":00")
          : "Данных недостаточно для выраженного пика";

      renderSummary([
        {
          title: "Итог периода",
          text: "За выбранный период создано " + formatInt(createdTotal) +
            " бронирований, из них " + formatInt(reservedTotal) +
            " попали в фактическую временную сетку. Текущее соотношение — " + formatPct(ratio) + "."
        },
        {
          title: "Пиковое фактическое время",
          text: timelinePeakValue > 0
            ? ("Наиболее нагруженный слот по фактическому времени: " + timelinePeakLabel + ", всего " + formatInt(timelinePeakValue) + " броней.")
            : "По фактическому времени в выбранном окне нет выраженного пика."
        },
        {
          title: "Когда чаще создают",
          text: createPeakValue > 0
            ? ("Наибольшая активность по созданию броней зафиксирована в слоте " + createPeakLabel + ", всего " + formatInt(createPeakValue) + " созданий.")
            : "По моменту создания броней в выбранном окне нет выраженного пика."
        },
        {
          title: "Часовая структура",
          text: peakHourValue > 0
            ? ("Чаще всего брони ставят на " + peakHourLabel + ":00. Это видно по почасовому распределению reservation_dt.")
            : "Почасовое распределение пока не формирует читаемый максимум."
        }
      ]);

      renderSegmentTable(segments.buckets || []);

      destroyChart(chartTimeline);
      chartTimeline = new Chart(ctxTimeline, {
        type: "line",
        data: {
          labels: labelsTimeline,
          datasets: [
            {
              label: "Создано",
              data: createdTimeline,
              borderColor: "#ffcc00",
              backgroundColor: "rgba(255,204,0,0.18)",
              tension: 0.28,
              fill: false,
              pointRadius: 2,
              pointHoverRadius: 4
            },
            {
              label: "Фактическое время",
              data: reservedTimeline,
              borderColor: "#2f80ed",
              backgroundColor: "rgba(47,128,237,0.18)",
              tension: 0.28,
              fill: false,
              pointRadius: 2,
              pointHoverRadius: 4
            }
          ]
        },
        options: {
          ...chartOptionsBase(),
          interaction: { mode: "index", intersect: false }
        }
      });

      destroyChart(chartSegments);
      chartSegments = new Chart(ctxSegments, {
        type: "doughnut",
        data: {
          labels: (segments.buckets || []).map(x => x.label),
          datasets: [{
            data: (segments.buckets || []).map(x => safeNum(x.count)),
            backgroundColor: [
              "#2f80ed",
              "#56ccf2",
              "#27ae60",
              "#f2c94c",
              "#f2994a",
              "#eb5757"
            ],
            borderColor: "#ffffff",
            borderWidth: 3,
            hoverOffset: 6
          }]
        },
        options: {
          responsive: true,
          maintainAspectRatio: false,
          plugins: {
            legend: {
              position: "bottom",
              labels: {
                color: "#475467",
                boxWidth: 12,
                padding: 16,
                font: { size: 12, weight: "600" }
              }
            },
            tooltip: {
              backgroundColor: "#101828",
              titleColor: "#ffffff",
              bodyColor: "#ffffff",
              callbacks: {
                label: function(ctx){
                  const i = ctx.dataIndex;
                  const bucket = segments.buckets?.[i] || {};
                  return " " + (bucket.label || "") + ": " + formatInt(bucket.count) + " (" + formatPct(bucket.pct) + ")";
                }
              }
            }
          }
        }
      });

      destroyChart(chartHourly);
      chartHourly = new Chart(ctxHourly, {
        type: "bar",
        data: {
          labels: labelsHour,
          datasets: [
            {
              label: "Создано",
              data: createdHourSeries,
              backgroundColor: "rgba(255,204,0,0.72)",
              borderRadius: 6
            },
            {
              label: "Фактическое время",
              data: reservedHourSeries,
              backgroundColor: "rgba(47,128,237,0.78)",
              borderRadius: 6
            }
          ]
        },
        options: {
          ...chartOptionsBase(),
          interaction: { mode: "index", intersect: false }
        }
      });

      setStatus("Обновлено: " + new Date().toLocaleString("ru-RU"));
    } catch(err){
      setStatus("Ошибка загрузки: " + (err && err.message ? err.message : String(err)));
    } finally {
      elReload.disabled = false;
    }
  }

  elReload.addEventListener("click", loadData);
  elPeriod.addEventListener("change", loadData);

  elPeriod.value = "today";
  loadData();
})();
